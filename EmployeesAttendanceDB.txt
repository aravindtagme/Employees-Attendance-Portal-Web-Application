-- 1. Create the Database
CREATE DATABASE EmployeeAttendanceDB;
GO
USE EmployeeAttendanceDB;
GO

-- 2. Create Admin Table for Login
CREATE TABLE AdminLogin (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Username NVARCHAR(50),
    Password NVARCHAR(50)
);

-- 3. Create Employees Table
CREATE TABLE Employees (
    EmpId INT PRIMARY KEY IDENTITY(1,1),
    EmpName NVARCHAR(100),
    Department NVARCHAR(50) -- IT, HR, or Helpdesk
);

-- 4. Create Attendance Table
CREATE TABLE Attendance (
    Id INT PRIMARY KEY IDENTITY(1,1),
    EmpId INT FOREIGN KEY REFERENCES Employees(EmpId),
    Status NVARCHAR(20),
    AttDate DATETIME DEFAULT GETDATE()
);
